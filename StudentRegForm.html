Enrolled Students

enrollment.js const form = document.getElementById("enrollment-form"); const table = document.getElementById("enrolled-students"); form.addEventListener("submit", function (event) { event.preventDefault(); const name = form.elements["name"].value; const email = form.elements["email"].value; const phone = form.elements["phone"].value; const major = form.elements["major"].value; const year = form.elements["year"].value; const row = table.insertRow(); const nameCell = row.insertCell(); const emailCell = row.insertCell(); const phoneCell = row.insertCell(); const majorCell = row.insertCell(); const yearCell = row.insertCell(); nameCell.textContent = name; emailCell.textContent = email; phoneCell.textContent = phone; majorCell.textContent = major; yearCell.textContent = year; form.reset(); });
